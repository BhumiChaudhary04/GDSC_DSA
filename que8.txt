class Solution {
public:
    bool isMonotonic(vector<int>& nums) {
        bool i=true;
        bool d=true;
        for(int i=0;i<nums.size()-1;++i)
        {
            if(nums[i]>nums[i+1]) i=false;
        
            if(nums[i]<nums[i+1]) d=false;
            
        }
        
        return i || d;
        

    }
};
